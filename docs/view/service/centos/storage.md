```bash
df -lh

# /dev/vda1        40G  4.0G   34G   11% /
# devtmpfs        911M     0  911M    0% /dev
# tmpfs           920M     0  920M    0% /dev/shm
# tmpfs           920M  420K  920M    1% /run
# tmpfs           920M     0  920M    0% /sys/fs/cgroup
# tmpfs           184M     0  184M    0% /run/user/0
```
